## Core Mechanics

### Primary Action
- **Type letters**: Enter 5-letter words (or variable length based on difficulty)
- **Submit guess**: Press Enter/Tap button to check word
- **Letter feedback**: Each letter colored to indicate correctness
  - **Green/Gold**: Correct letter in correct position
  - **Yellow/Orange**: Correct letter in wrong position  
  - **Gray/Dark**: Letter not in word at all

### Win Conditions
- Guess the target word within 6 attempts
- Optional challenge: Guess in minimum attempts (bonus points)

### Fail Conditions
- Use all 6 attempts without finding the word
- Word is revealed, but no points/stars earned

### Progression System
- **Daily word**: New word each day (like original Wordle)
- **Category progression**: Unlock new term categories as you play
- **Streak tracking**: Consecutive days played
- **Difficulty tiers**:
  - Beginner: 4-letter common terms (DRILL, CORE, BLADE)
  - Standard: 5-letter industry terms (GROUT, ARBOR, KERF)
  - Expert: 6-letter technical terms (BRAZED, TURBO, MATRIX)
  - Master: 7+ letter terms (DIAMOND, SEGMENT, COOLANT)

### Game Loop
1. Player sees empty grid (6 rows of 5 boxes)
2. Player types first word guess
3. Submit - each letter receives color feedback
4. Use feedback to inform next guess
5. Repeat until word found or attempts exhausted
6. Victory screen with stats, streak, share option

class WordManager {
  constructor(scene) {
    this.scene = scene;
    this.wordLength = 5;
    this.maxAttempts = 6;
    this.currentAttempt = 0;
    this.currentGuess = '';
    this.targetWord = '';
    this.validWords = [];
    this.guessedWords = [];
    
    this.loadWordDatabase();
    this.selectDailyWord();
  }
  
  loadWordDatabase() {
    // Load from JSON asset
    const data = this.scene.cache.json.get('word_database');
    this.categories = data.categories;
    this.validWords = data.validWords; // Accepted guesses
    this.targetWords = data.targetWords; // Possible answers
  }
  
  selectDailyWord() {
    // Daily word based on date (like Wordle)
    const today = new Date();
    const startDate = new Date('2024-01-01');
    const dayIndex = Math.floor((today - startDate) / (1000 * 60 * 60 * 24));
    
    // Use current category/difficulty
    const categoryWords = this.getWordsForCurrentLevel();
    this.targetWord = categoryWords[dayIndex % categoryWords.length].toUpperCase();
  }
  
  getWordsForCurrentLevel() {
    // Get words based on player's unlocked level
    const level = this.scene.gameState.currentLevel;
    if (level <= 10) return this.categories.basic_tools;
    if (level <= 20) return this.categories.specifications;
    if (level <= 30) return this.categories.materials;
    if (level <= 40) return this.categories.applications;
    return this.categories.advanced_technical;
  }
  
  addLetter(letter) {
    if (this.currentGuess.length < this.wordLength) {
      this.currentGuess += letter;
      this.scene.updateGuessDisplay(this.currentAttempt, this.currentGuess);
    }
  }
  
  removeLetter() {
    if (this.currentGuess.length > 0) {
      this.currentGuess = this.currentGuess.slice(0, -1);
      this.scene.updateGuessDisplay(this.currentAttempt, this.currentGuess);
    }
  }
  
  submitGuess() {
    if (this.currentGuess.length !== this.wordLength) {
      this.scene.showError('Not enough letters');
      return;
    }
    
    if (!this.isValidWord(this.currentGuess)) {
      this.scene.showError('Not in word list');
      this.scene.shakeRow(this.currentAttempt);
      return;
    }
    
    // Evaluate guess
    const feedback = this.evaluateGuess(this.currentGuess);
    this.scene.revealGuess(this.currentAttempt, this.currentGuess, feedback);
    this.guessedWords.push({word: this.currentGuess, feedback});
    
    // Update keyboard colors
    this.scene.updateKeyboard(this.currentGuess, feedback);
    
    // Check win/lose
    if (this.currentGuess === this.targetWord) {
      this.scene.time.delayedCall(1500, () => {
        this.scene.showVictory(this.currentAttempt + 1);
      });
    } else if (this.currentAttempt >= this.maxAttempts - 1) {
      this.scene.time.delayedCall(1500, () => {
        this.scene.showDefeat(this.targetWord);
      });
    } else {
      this.currentAttempt++;
      this.currentGuess = '';
    }
  }
  
  isValidWord(word) {
    return this.validWords.includes(word.toUpperCase());
  }
  
  evaluateGuess(guess) {
    const feedback = [];
    const targetLetters = this.targetWord.split('');
    const guessLetters = guess.toUpperCase().split('');
    const used = new Array(this.wordLength).fill(false);
    
    // First pass: Mark correct positions (green)
    for (let i = 0; i < this.wordLength; i++) {
      if (guessLetters[i] === targetLetters[i]) {
        feedback[i] = 'correct';
        used[i] = true;
      }
    }
    
    // Second pass: Mark wrong positions (yellow)
    for (let i = 0; i < this.wordLength; i++) {
      if (feedback[i]) continue; // Already marked correct
      
      const letterIndex = targetLetters.findIndex((letter, idx) => 
        letter === guessLetters[i] && !used[idx]
      );
      
      if (letterIndex !== -1) {
        feedback[i] = 'present';
        used[letterIndex] = true;
      } else {
        feedback[i] = 'absent';
      }
    }
    
    return feedback;
  }
  
  getWordDefinition(word) {
    const data = this.scene.cache.json.get('word_definitions');
    return data[word] || {
      definition: 'Industry term related to core drilling and diamond tools.',
      example: 'Used in professional applications.',
      product: null
    };
  }
}

class WordGrid {
  constructor(scene, rows, cols) {
    this.scene = scene;
    this.rows = rows;
    this.cols = cols;
    this.tiles = [];
    
    this.createGrid();
  }
  
  createGrid() {
    const tileSize = 70;
    const gap = 8;
    const startX = 400 - (this.cols * (tileSize + gap)) / 2;
    const startY = 100;
    
    for (let row = 0; row < this.rows; row++) {
      this.tiles[row] = [];
      for (let col = 0; col < this.cols; col++) {
        const x = startX + col * (tileSize + gap);
        const y = startY + row * (tileSize + gap);
        
        const tile = this.createTile(x, y);
        this.tiles[row][col] = tile;
      }
    }
  }
  
  createTile(x, y) {
    const tileSize = 70;
    
    // Container
    const container = this.scene.add.container(x, y);
    
    // Background
    const bg = this.scene.add.rectangle(0, 0, tileSize, tileSize, 0x3A3A3C);
    bg.setStrokeStyle(2, 0x565758);
    
    // Letter text
    const text = this.scene.add.text(0, 0, '', {
      fontSize: '36px',
      fontFamily: 'Montserrat, Arial',
      fontStyle: 'bold',
      color: '#ffffff'
    }).setOrigin(0.5);
    
    container.add([bg, text]);
    
    return {container, bg, text, letter: '', state: 'empty'};
  }
  
  updateTile(row, col, letter, animate = false) {
    const tile = this.tiles[row][col];
    tile.letter = letter;
    tile.text.setText(letter);
    
    if (animate && letter) {
      // Pop animation
      tile.container.setScale(0.8);
      this.scene.tweens.add({
        targets: tile.container,
        scale: 1,
        duration: 100,
        ease: 'Back.easeOut'
      });
    }
    
    if (!letter) {
      // Empty tile
      tile.bg.setFillStyle(0x3A3A3C);
      tile.state = 'empty';
    } else {
      // Filled but not revealed
      tile.bg.setFillStyle(0x565758);
      tile.state = 'filled';
    }
  }
  
  revealTile(row, col, letter, feedback, delay = 0) {
    const tile = this.tiles[row][col];
    
    this.scene.time.delayedCall(delay, () => {
      // Flip animation
      this.scene.tweens.add({
        targets: tile.container,
        scaleY: 0,
        duration: 200,
        ease: 'Quad.easeIn',
        onComplete: () => {
          // Change color during flip
          const colors = {
            correct: 0x538D4E,   // Green
            present: 0xB59F3B,   // Yellow
            absent: 0x3A3A3C     // Gray
          };
          
          tile.bg.setFillStyle(colors[feedback]);
          tile.state = feedback;
          
          // Flip back
          this.scene.tweens.add({
            targets: tile.container,
            scaleY: 1,
            duration: 200,
            ease: 'Quad.easeOut'
          });
          
          // Sound
          this.scene.sound.play('tile_flip', {volume: 0.3});
          
          // Extra effect for correct
          if (feedback === 'correct') {
            this.scene.tweens.add({
              targets: tile.container,
              scale: 1.1,
              duration: 100,
              yoyo: true
            });
          }
        }
      });
    });
  }
  
  shakeRow(row) {
    this.tiles[row].forEach(tile => {
      this.scene.tweens.add({
        targets: tile.container,
        x: tile.container.x - 5,
        duration: 50,
        yoyo: true,
        repeat: 2,
        ease: 'Quad.easeInOut'
      });
    });
    this.scene.sound.play('error', {volume: 0.5});
  }
  
  bounceRow(row, callback) {
    // Victory animation - bounce each tile in sequence
    this.tiles[row].forEach((tile, index) => {
      this.scene.time.delayedCall(index * 100, () => {
        this.scene.tweens.add({
          targets: tile.container,
          y: tile.container.y - 20,
          duration: 300,
          yoyo: true,
          ease: 'Quad.easeOut',
          onComplete: () => {
            if (index === this.cols - 1 && callback) {
              callback();
            }
          }
        });
      });
    });
  }
}

class OnScreenKeyboard {
  constructor(scene) {
    this.scene = scene;
    this.keys = {};
    this.keyStates = {}; // Track color of each key
    
    this.createKeyboard();
  }
  
  createKeyboard() {
    const rows = [
      ['Q', 'W', 'E', 'R', 'T', 'Y', 'U', 'I', 'O', 'P'],
      ['A', 'S', 'D', 'F', 'G', 'H', 'J', 'K', 'L'],
      ['ENTER', 'Z', 'X', 'C', 'V', 'B', 'N', 'M', 'BACK']
    ];
    
    const keyWidth = 45;
    const keyHeight = 60;
    const gap = 6;
    const startY = 550;
    
    rows.forEach((row, rowIndex) => {
      const rowWidth = row.reduce((sum, key) => {
        const width = key.length > 1 ? keyWidth * 1.5 : keyWidth;
        return sum + width + gap;
      }, -gap);
      
      const startX = 400 - rowWidth / 2;
      let x = startX;
      
      row.forEach(letter => {
        const width = letter.length > 1 ? keyWidth * 1.5 : keyWidth;
        const y = startY + rowIndex * (keyHeight + gap);
        
        const key = this.createKey(x + width / 2, y, width, keyHeight, letter);
        this.keys[letter] = key;
        this.keyStates[letter] = 'unused';
        
        x += width + gap;
      });
    });
  }
  
  createKey(x, y, width, height, letter) {
    const container = this.scene.add.container(x, y);
    
    // Background
    const bg = this.scene.add.rectangle(0, 0, width, height, 0x818384);
    bg.setStrokeStyle(1, 0x565758);
    bg.setInteractive(new Phaser.Geom.Rectangle(-width/2, -height/2, width, height), Phaser.Geom.Rectangle.Contains);
    
    // Letter text
    const text = this.scene.add.text(0, 0, letter, {
      fontSize: letter.length > 1 ? '14px' : '20px',
      fontFamily: 'Arial',
      fontStyle: 'bold',
      color: '#ffffff'
    }).setOrigin(0.5);
    
    container.add([bg, text]);
    
    // Click handler
    bg.on('pointerdown', () => {
      if (letter === 'ENTER') {
        this.scene.wordManager.submitGuess();
      } else if (letter === 'BACK') {
        this.scene.wordManager.removeLetter();
      } else {
        this.scene.wordManager.addLetter(letter);
      }
      
      // Press animation
      this.scene.tweens.add({
        targets: container,
        scale: 0.95,
        duration: 50,
        yoyo: true
      });
    });
    
    return {container, bg, text, letter};
  }
  
  updateKey(letter, feedback) {
    const key = this.keys[letter];
    if (!key) return;
    
    // Priority: correct > present > absent
    const currentState = this.keyStates[letter];
    if (currentState === 'correct') return; // Don't downgrade
    if (currentState === 'present' && feedback === 'absent') return;
    
    const colors = {
      correct: 0x538D4E,
      present: 0xB59F3B,
      absent: 0x3A3A3C,
      unused: 0x818384
    };
    
    this.scene.tweens.add({
      targets: key.bg,
      fillColor: colors[feedback],
      duration: 300
    });
    
    this.keyStates[letter] = feedback;
  }
}

class DiamondWordsGame extends Phaser.Scene {
  create() {
    // Background
    this.add.image(400, 300, 'background');
    
    // Title
    this.add.text(400, 30, 'DIAMOND WORDS', {
      fontSize: '32px',
      fontFamily: 'Montserrat',
      fontStyle: 'bold',
      color: '#ffffff'
    }).setOrigin(0.5);
    
    // Initialize systems
    this.wordManager = new WordManager(this);
    this.grid = new WordGrid(this, 6, 5);
    this.keyboard = new OnScreenKeyboard(this);
    
    // Physical keyboard support
    this.input.keyboard.on('keydown', (event) => {
      if (event.key === 'Enter') {
        this.wordManager.submitGuess();
      } else if (event.key === 'Backspace') {
        this.wordManager.removeLetter();
      } else if (/^[a-zA-Z]$/.test(event.key)) {
        this.wordManager.addLetter(event.key.toUpperCase());
      }
    });
  }
  
  updateGuessDisplay(row, guess) {
    for (let col = 0; col < 5; col++) {
      const letter = guess[col] || '';
      this.grid.updateTile(row, col, letter, true);
    }
  }
  
  revealGuess(row, guess, feedback) {
    for (let col = 0; col < guess.length; col++) {
      const letter = guess[col];
      this.grid.revealTile(row, col, letter, feedback[col], col * 150);
      
      // Update keyboard after last tile
      if (col === guess.length - 1) {
        this.time.delayedCall((col + 1) * 150 + 200, () => {
          this.updateKeyboard(guess, feedback);
        });
      }
    }
  }
  
  updateKeyboard(guess, feedback) {
    for (let i = 0; i < guess.length; i++) {
      this.keyboard.updateKey(guess[i], feedback[i]);
    }
  }
  
  shakeRow(row) {
    this.grid.shakeRow(row);
  }
  
  showVictory(attempts) {
    // Bounce animation
    this.grid.bounceRow(attempts - 1, () => {
      this.showResultsScreen(true, attempts);
    });
    this.sound.play('victory');
  }
  
  showDefeat(correctWord) {
    this.time.delayedCall(500, () => {
      this.showResultsScreen(false, 6, correctWord);
    });
    this.sound.play('defeat');
  }
  
  showResultsScreen(won, attempts, correctWord = null) {
    // Modal overlay
    const overlay = this.add.rectangle(400, 300, 800, 600, 0x000000, 0.8);
    
    // Results panel
    const panel = this.add.container(400, 300);
    const bg = this.add.rectangle(0, 0, 500, 400, 0x2C3E50);
    bg.setStrokeStyle(4, 0x0047AB);
    
    // Title
    const title = won ? 'VICTORY!' : 'BETTER LUCK NEXT TIME';
    const titleText = this.add.text(0, -150, title, {
      fontSize: '32px',
      fontFamily: 'Montserrat',
      fontStyle: 'bold',
      color: '#F1C40F'
    }).setOrigin(0.5);
    
    // Word info
    const word = correctWord || this.wordManager.targetWord;
    const wordInfo = this.wordManager.getWordDefinition(word);
    
    const wordText = this.add.text(0, -100, word, {
      fontSize: '48px',
      fontFamily: 'Montserrat',
      fontStyle: 'bold',
      color: '#ffffff'
    }).setOrigin(0.5);
    
    const definition = this.add.text(0, -50, wordInfo.definition, {
      fontSize: '16px',
      fontFamily: 'Arial',
      color: '#ECF0F1',
      align: 'center',
      wordWrap: {width: 450}
    }).setOrigin(0.5);
    
    const example = this.add.text(0, 30, `Example: ${wordInfo.example}`, {
      fontSize: '14px',
      fontFamily: 'Arial',
      color: '#95A5A6',
      align: 'center',
      wordWrap: {width: 450}
    }).setOrigin(0.5);
    
    // Stats
    if (won) {
      const stats = this.add.text(0, 80, `Solved in ${attempts}/6 attempts`, {
        fontSize: '18px',
        color: '#F1C40F'
      }).setOrigin(0.5);
      panel.add(stats);
    }
    
    // Buttons
    const nextButton = this.createButton(0, 140, 'Next Word', () => {
      this.scene.restart();
    });
    
    const shareButton = this.createButton(0, 190, 'Share Result', () => {
      this.shareResult(won, attempts);
    });
    
    panel.add([bg, titleText, wordText, definition, example, nextButton, shareButton]);
    
    // Animate in
    panel.setScale(0);
    this.tweens.add({
      targets: panel,
      scale: 1,
      duration: 300,
      ease: 'Back.easeOut'
    });
  }
